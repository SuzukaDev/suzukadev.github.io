{{- /* Makes a wavy text Arguments: text = (string) Shortcut freq = (float/int) The frequency of the wave. Default: .1 amp = (float/int) Amplitude of the wave. Default: 20 time = (float/int) Time for the effect. Default: 1 Usage: {{< text_wavy "helloooo!" >}} {{< text_wavy text="helloooo!" amp=10 >}} */ -}} {{- /* REFACTOR text_wavy */ -}} {{- $text_input := or (.Get "text") (.Get 0) -}} {{- $chars := split $text_input "" -}} {{- /* {{- $frequency := (.Get 1) | default .2 | mul -1.0}} */ -}} {{- /* {{- $rawFrequency := (.Get 1) | default .2 -}} */ -}} {{- $rawFrequency := float ((or (.Get "freq") (.Get 1)) | default .1) -}} {{- $frequency := mul (math.Abs $rawFrequency) -1 -}} {{- /* {{- $amp := (.Get "amp") | default 20 -}} */ -}} {{- $amp := float ((or (.Get "amp") (.Get 2)) | default 10) -}} {{- $half_amp := div $amp 2.0 -}} {{- /* {{$max_y := 10}} {{$min_y := mul $max_y -1}} */ -}} {{- $max_y := $half_amp -}} {{- $min_y := mul $half_amp -1.0 -}} {{- /* {{$animation_time := (.Get "time") | default 1}} */ -}} {{- $animation_time := float ((.Get "time") | default 1) -}} {{- /* {{- range $index, $char := $chars -}} 
{{$char}}
{{end}} */ -}} {{- /*
*/ -}} {{- $output := "" -}} {{- range $index, $char := $chars -}} {{- $delay := mul $index ($frequency) -}} {{- $delay = float $delay -}} {{- /* NOTE THIS ADDS WHITESPACES! */ -}} {{- /* INFO https://discourse.gohugo.io/t/hugo-range-function-adds-extra-whitespace-between-spans/33219 */ -}} {{- /* {{- $char -}} */ -}} {{- if eq $char " " -}} {{- /* {{- $output = printf "%s$nbsp;" ($output | safeHTMLAttr) -}} */ -}} {{- /* {{- $output = printf "%s " ($output | safeHTMLAttr) -}} */ -}} {{- /* {{- $output = printf "%s " $output -}} */ -}} {{- $output = printf "%s " $output -}} {{- else -}} {{- $output = printf "%s%s" $output $animation_time ($delay) $min_y $max_y $char -}} {{- end -}} {{- end -}} {{- /* To avoid whitespaces in HTML */ -}} {{- $output | safeHTML -}}
