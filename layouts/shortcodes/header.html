{{- /* Displays a header with align options. Arguments: 1 text = (String). Text of the header. 2 h = (int) Type of header for the tile (1 for h1, 2 for h2, 3 for h3, etc.). Default: 2 (h2) 3 align = (String) Text align of the header (options: "left", "right", "center"). Default: "center" 4 class = (String) CSS class for the header. Default: "" Use example: {{< header "my-first-article, another-article" >}} {{< header "my-first-article, another-article" "My favourite articles" 3 "left" >}} {{< header title="my-first-article, another-article" title="My favourite articles" align="right" class="bloom-shadow" >}} */ -}} {{- $title := or (.Get 0) (.Get "text") (.Get "t") -}} {{- $header_number := or (.Get 1) (.Get "h") | default 2 -}} {{- $text_align := or (.Get 2) (.Get "align") | default "center" -}} {{- $css_class := or (.Get 3) (.Get "class") | default "" -}} {{- $title = $title | markdownify}} {{- $css_class_attr := "" -}} {{- if $css_class -}} {{- $css_class_attr = (printf "class='%s'" $css_class) | safeHTMLAttr -}} {{- end -}} {{- $header_tag := printf "h%d" $header_number -}} {{- $flex_align := "center" -}} {{- if eq $text_align "right" -}} {{- $flex_align = "flex-end" -}} {{- else if eq $text_align "left" -}} {{- $flex_align = "flex-start" -}} {{- end -}} {{- $style := (printf "text-align:%s; align-self:%s;" $text_align $flex_align) | safeCSS -}} {{- $header := printf "<%s %s style='%s'>%s" $header_tag $css_class_attr $style $title $header_tag -}} {{- $header | safeHTML -}}
